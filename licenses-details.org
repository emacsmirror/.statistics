* Packages that don't specify license and are available from Melpa

The authors have been contacted.

#+BEGIN_SRC emacs-lisp
  (epkg-with-org-header ("Package" "Class" "Link")
    (epkg-sql [:select :distinct [packages:name packages:class packages:repopage]
               :from [packages melpa-recipes]
               :where (and (= packages:name melpa-recipes:epkg-package)
                           (= packages:license "pending"))
               :group-by packages:repopage
               :order-by [(asc packages:name)]]))
