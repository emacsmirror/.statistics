    (-filter (pcase-lambda (`(,name ,class))
               (and (not (assoc name emir-pending-packages))
                    (not (assoc name emir--preferred-builtins))
                    (or (not (eq class 'core))
                        (epkg-builtin-package-p (epkg name)))))
             (epkg-sql [:select [name class epkg-package]
                        :from gelpa-recipes
                        :where (or (isnull epkg-package)
                                   (= class 'core))

** Diverging repositories
#+NAME: melpa-diverging-repositories
#+BEGIN_SRC emacs-lisp
  (epkg-with-org-header ("Package" "Mirror" "Melpa")
    (-keep
     (-lambda ((name melpa-url))
       (and (not (or (equal name "yatex")
                     (string-prefix-p "git@github.com:emacsmirror/" melpa-url)
                     (string-prefix-p "git@github.com:emacsattic/"  melpa-url)))
            (list name
                  (epkg-org-link  name)
                  (melpa-org-link name))))
     (epkg-sql [:select :distinct [packages:name melpa-recipes:url]
                :from [packages melpa-recipes]
                :where    (= melpa-recipes:epkg-package packages:name)
                :and      (= melpa-recipes:epkg-package melpa-recipes:name)
                :and :not (= packages:url melpa-recipes:url)
                :and :not (= packages:class 'minority)
                :and :not (= packages:class 'subtree)
                :and :not (= packages:class 'builtin)
                :order-by [(asc packages:name)]])))

** Diverging types
#+NAME: melpa-diverging-types
#+BEGIN_SRC emacs-lisp
  (epkg-with-org-header ("Package" "Class" "Melpa" "Reason")
    (cl-sort
     (-keep
      (pcase-lambda (`(,name ,class ,fetcher ,repo))
        (and (not (pcase (list class fetcher)
                    (`(subtree  github) t)
                    (`(subtree  git)    t)))
             (pcase (list class fetcher)
               (`(orphaned github)
                (list name class "ORPHAN" "non-git/hg"))
               ((and `(file github)
                     (guard (string-prefix-p "emacsmirror/" repo)))
                (list name class "MIRROR" "non-git/hg"))
               ((and `(shelved github)
                     (guard (string-prefix-p "emacsattic/" repo)))
                (list name class "ATTIC" "obsolete"))
               (_
                (list name class fetcher
                      (cadr (assoc name emir--diverging-reason)))))))
      (epkg-sql [:select :distinct [packages:name
                                    packages:class
                                    melpa-recipes:class
                                    melpa-recipes:repo]
                 :from [packages melpa-recipes]
                 :where    (= melpa-recipes:epkg-package packages:name)
                 :and      (= melpa-recipes:epkg-package melpa-recipes:name)
                 :and :not (= melpa-recipes:class        packages:class)
                 :order-by [(asc packages:name)]]))
     #'string< :key #'cl-cadddr))
#+END_SRC
