* Features provided by multiple packages
#+NAME: conflict
#+BEGIN_SRC emacs-lisp
  (let (alist)
    (dolist (name (epkgs 'name))
      (dolist (feature (epkg-sql [:select feature :from provided
                                  :where (and (= package $s1)
                                              (isnull drop))]
                                 name))
        (setq feature (car feature))
        (-if-let (elt (assq feature alist))
            (push name (cdr elt))
          (cl-pushnew (list feature name) alist))))
    (epkg-with-org-header ("Feature" "Package" "Shelved")
      (mapcar (-lambda ((feature package))
                (list feature package
                      (epkg-shelved-package-p (epkg package))))
              (-mapcat (-lambda ((feature . providers))
                         (and (> (length providers) 1)
                              (list (list feature providers))
                              (--map (list feature it)
                                     (sort providers #'string<))))
                       (cl-sort alist #'string< :key #'car)))))
#+END_SRC
#+RESULTS: conflict

* Packages with unsatisfied dependencies
#+NAME: unsatisfied
#+BEGIN_SRC emacs-lisp
  (epkg-with-org-header ("Package" "Class" "Feature" "Shelved")
    (-mapcat
     (-lambda ((name class))
       (let (alist)
	 (-each (epkg-sql [:select feature :from required
                           :where (and (= package $s1)
                                       (isnull drop))]
                          name)
           (-lambda ((feature))
             (-if-let (provider (epkg-provided-by feature))
		 (when (epkg-sql [:select class :from packages
                                  :where (and (= name $s1)
                                              (= class 'shelved))]
				 provider)
                   (push (list feature provider) alist))
               (push (list feature nil) alist))))
	 (mapcar (-lambda ((feature provider))
                   (list name class feature provider))
		 (cl-sort alist #'string< :key #'car))))
     (epkgs [name class] 'epkg-mirrored-package--eieio-childp)))
#+END_SRC

* Insecure packages

All packages from the Emacswiki are completely insecure because
anyone can edit any pacakge without having to perform any sort of
attack.  It is possible to do so by design.  Those packages are
not listed here.

Also not listed here are shelved packages because for many of these
packages it is not known whether they used to be updated over a secure
connection or not.

Some packages that are manually imported into the Emacsorphanage and
then into the Emacsmirror from there are probably insecure too.

Note: I last checked 2017-12-07 whether any of these files are now
available over https.

#+BEGIN_SRC emacs-lisp
  (epkg-with-org-header ("Package" "Class" "Year" "M" "Dep" "Url")
    (-keep (pcase-lambda (`(,name ,class ,updated ,url))
             (and (or (not url)
                      (string-prefix-p "http://" url)
                      (string-prefix-p "git://" url))
                  (list name class
                        (substring updated 0 4)
                        (and (melpa-get name) t)
                        (let ((deps (epkg-reverse-dependencies (epkg name))))
                          (and deps (length deps)))
                        url)))
           (epkg-sql [:select [name class updated url]
                      :from packages
                      :where class :in $v1
                      :order-by [(desc class) (asc url) (asc name)]]
                     (closql-where-class-in
                      'epkg-mirrored-package--eieio-childp))))
#+END_SRC
