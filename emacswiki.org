#+BEGIN_SRC emacs-lisp
  (epkg-with-org-header ("Dependee" "Author" "Depender" "Type" "Fetcher" "Author")
    (-mapcat
     (-lambda ((dependee _dependee-type))
       (let ((dependee-pkg (epkg dependee)))
         (-mapcat (-lambda ((depender . features))
                    (let ((depender-pkg  (epkg depender))
                          (depender-type (caar (epkg-sql [:select class
                                                          :from melpa-recipes
                                                          :where (= name $s1)]
                                                         depender))))
                      (and depender-type
                           (not (eq depender-type 'wiki))
                           ;; (not (epkg-wiki-package-p depender-pkg))
                           `((,dependee
                              ,(caar (oref dependee-pkg authors))
                              ,depender
                              ,(if (-any #'symbolp features) "hard" "soft")
                              ,depender-type
                              ,(caar (oref depender-pkg authors)))))))
                  (epkg-reverse-dependencies dependee))))
     ;; (epkgs [name class] 'epkg-wiki-package-p))
     (epkg-sql [:select :distinct [packages:name packages:class]
                :from [packages melpa-recipes]
                :where (= packages:name melpa-recipes:epkg-package)
                :and   (= packages:class 'wiki)
                :order-by [(asc packages:name)
                           (asc melpa-recipes:name)]])))
