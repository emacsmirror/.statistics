    (-mapcat
     (-lambda ((dependee _dependee-type))
       (let ((dependee-pkg (epkg dependee)))
	 (-mapcat (-lambda ((depender . _))
		    (let ((depender-pkg  (epkg depender))
			  (depender-type (caar (epkg-sql [:select fetcher
							  :from melpa-recipes
							  :where (= name $s1)]
							 depender))))
		      (and depender-type
			   (not (eq depender-type 'wiki))
			   ;; (not (epkg-wiki-package-p depender-pkg))
			   (list (list dependee
				       (or (caar (oref dependee-pkg authors))"")
				       depender
				       ;; (epkg-type depender-pkg)
				       depender-type
				       (or (caar (oref depender-pkg authors))"")
				       )))
		      ))
		  (epkg-reverse-dependencies dependee))))
     ;; (epkgs [name class] 'epkg-wiki-package-p))
     (epkg-sql [:select :distinct [packages:name packages:class]
		:from [packages melpa-recipes]
		:where (= packages:name melpa-recipes:closql-id)
		:and   (= packages:class 'wiki)
		:order-by [(asc packages:name)
                           (asc melpa-recipes:name)]])))
